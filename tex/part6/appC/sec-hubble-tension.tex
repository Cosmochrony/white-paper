\subsection{Evolution of the Hubble Parameter and the Hubble Tension}
\label{app:hubble_tension}

At the homogeneous background level, $a(t) \propto \chi(t)$ and
$H(t) = \dot{\chi}/\chi$.
In the idealized limit $\dot{\chi} = c$, one obtains
$H(t) = c/\chi(t)$.

\paragraph{Illustrative $H(z)$ profile.}
Using $1+z = \chi(t_0)/\chi(t)$, the simplest background closure gives
\begin{equation}
  H_{\chi}(z)
  = H_0\,(1+z)\,\sqrt{1-\Omega_{\chi}}
  \quad(\text{illustrative}),
  \label{eq:Hz_cosmochrony_illustrative}
\end{equation}
to be compared with
\begin{equation}
  H_{\Lambda\mathrm{CDM}}(z)
  = H_0
    \sqrt{\Omega_m(1+z)^3
      + \Omega_r(1+z)^4
      + \Omega_\Lambda}.
  \label{eq:Hz_LCDM}
\end{equation}

\paragraph{Relaxation budget.}
The dimensionless parameter
\begin{equation}
  \Omega_\chi \equiv \langle \beta^2 \rangle,
  \quad
  \beta \equiv \frac{|\nabla \chi|}{c},
  \label{eq:chi-relaxation}
\end{equation}
measures the fraction of relaxation capacity stored in spatial gradients.
The effective global expansion rate is then
$\bar{H} = (c/\chi)\sqrt{1 - \Omega_\chi}$.

\paragraph{Local expansion and the Hubble tension.}
For a region with density contrast~$\delta$, a minimal mean-field
closure gives
\begin{equation}
  H_{\mathrm{loc}}
  = \bar{H}
    \sqrt{
      \frac{1 - \Omega_\chi(1+\delta)}
           {1 - \Omega_\chi}
    }.
  \label{eq:hubble_local}
\end{equation}
For $\Omega_\chi \approx 0.3$ and a KBC-void underdensity
$\delta \approx -0.4$, one finds
$H_{\mathrm{loc}}/\bar{H} \approx 1.08$, comparable to the observed
tension.
Structural consistency imposes
$\Delta H / H \in [3.5\%,\,9.5\%]$ at $z \sim 1$.

\paragraph{Parameter degeneracies.}
The effective gravitational coupling constrains
\begin{equation}
  K_0\,\chi_c^2
  \sim \frac{c^4}{16\pi G},
  \label{eq:K0_chic_deg}
\end{equation}
so that $K_0$ and $\chi_c$ cannot be fixed independently without
additional input.
