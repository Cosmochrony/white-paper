\subsection{Strong-Field Constitutive Coupling Near a
Schwarzschild Black Hole}
\label{app:keff_schwarzschild}

The effective constitutive relation:
\begin{equation}
  K_{\mathrm{eff}}
  = K_0\exp\!\left(
      -\frac{(\Delta\chi)^2}{\chi_c^2}
    \right).
  \label{eq:keff_constitutive}
\end{equation}

The operational lapse-like factor:
\begin{equation}
  N(r)
  \equiv
  \frac{\mathcal{D}_{\mathrm{loc}}\chi(r)}
       {\mathcal{D}_0\chi},
  \quad 0 < N(r) \le 1.
  \label{eq:lapse_def}
\end{equation}

Matching to the Schwarzschild metric (purely effective):
\begin{equation}
  ds^2 = -f(r)c^2 dt^2 + f(r)^{-1}dr^2
    + r^2 d\Omega^2,
  \quad
  f(r) = 1 - \frac{r_s}{r}.
  \label{eq:schwarzschild_standard}
\end{equation}
\begin{equation}
  N(r)^2 = f(r) = 1 - \frac{r_s}{r}.
  \label{eq:lapse_schwarzschild}
\end{equation}

Minimal identification
$K_{\mathrm{eff}}(r)/K_0 \equiv N(r)^2$:
\begin{equation}
  K_{\mathrm{eff}}(r)/K_0
  \equiv N(r)^2.
  \label{eq:keff_lapse_ansatz}
\end{equation}
Yielding the strong-field profile:
\begin{equation}
  K_{\mathrm{eff}}(r)
  = K_0\!\left(1 - \frac{r_s}{r}\right),
  \quad r > r_s.
  \label{eq:keff_schwarzschild_profile}
\end{equation}

Inverting Eq.~\eqref{eq:keff_constitutive}:
\begin{equation}
  \frac{(\Delta\chi(r))^2}{\chi_c^2}
  = -\ln\!\left(1 - \frac{r_s}{r}\right).
  \label{eq:deltachi_schwarzschild}
\end{equation}
As $r \to r_s^+$:
\begin{equation}
  \Delta\chi(r)
  \sim \chi_c
    \sqrt{-\ln\!\left(
      1 - \frac{r_s}{r}
    \right)}.
  \label{eq:deltachi_horizon_asymptotic}
\end{equation}
This logarithmic divergence signals that the hydrodynamic
parametrization exceeds its domain of validity near the effective
horizon.
A Schwarzschild horizon corresponds to vanishing relaxation
conductivity ($K_{\mathrm{eff}} \to 0$), not a fundamental spacetime
singularity.
