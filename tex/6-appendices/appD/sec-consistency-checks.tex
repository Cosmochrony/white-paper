\subsection{Order-of-Magnitude Consistency Checks}
\label{sec:consistency_checks}

\begin{enumerate}
  \item \textbf{Electron mass scale.}
    The lowest stability eigenvalue must satisfy
    $\lambda_1 \sim (m_e c^2/\hbar_{\mathrm{eff}})^2
      \sim 10^{41}\,\mathrm{s^{-2}}$,
    implying $K_0 \sim 10^{31}\,\mathrm{m}^{-2}$ for
    $a \sim 10^{-15}\,\mathrm{m}$.
  \item \textbf{Correlation scale.}
    Electroweak compatibility requires
    $\chi_c \lesssim \hbar c/v
      \sim 10^{-18}\,\mathrm{m}$.
  \item \textbf{Absence of fine-tuning.}
    Viable regimes are defined by
    $K_0 a^2 \gg 1$,
    $\chi_c \lesssim 10^{-18}\,\mathrm{m}$, and
    $K_0 \lesssim c^2/a^2$.
\end{enumerate}
